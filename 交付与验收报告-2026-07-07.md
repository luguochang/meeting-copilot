## 4. 部署与验收命令

```bash
# 1. 安装依赖
cd code/web_mvp/backend
python3 -m pip install -e .  # fastapi/uvicorn[standard]/pydantic/httpx/dotenv/structlog/websockets

# 2. 配置 .env（从 .env.example 复制，填 LLM 网关）
cp ../../.env.example ../../.env  # 编辑填入 LLM_GATEWAY_*

# 3. 跑测试
PYTHONPATH=".:../../core" python3 -m pytest -q          # 162 passed
cd ../../core && python3 -m pytest -q                    # 34 passed
cd ../../asr_bakeoff && python3 -m pytest -q             # 18 passed

# 4. 启动后端
cd ../web_mvp/backend
PYTHONPATH=".:../../core" MEETING_COPILOT_DATA_DIR=/tmp/mc \
  uvicorn meeting_copilot_web_mvp.app:app --port 8000

# 5. 浏览器验收：http://127.0.0.1:8000/workbench
#    - 加载样本会议 → 实时订阅 → 生成建议卡片（需 LLM）→ 方案考量（需 LLM）→ 删除会话

# 6. 真实 ASR 管线（mic_capture --simulate 干净 wav → WS → sherpa sidecar）
python3 -m meeting_copilot_web_mvp.mic_capture --session-id real \
  --ws-url ws://127.0.0.1:8000/live/asr/stream/ws/real \
  --simulate ../../asr_runtime/outputs/simulated-release-review.16k.wav

# 7. Tauri 桌面（需 rust + tauri CLI）
cd code/desktop_tauri/src-tauri && cargo check
cargo tauri build  # 产出 .app/.dmg（需签名）

# 8. metrics
curl http://127.0.0.1:8000/metrics
```
